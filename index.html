function sentence(List) {
  let my_arr = [];
  for (let value of List) {
      let arr_item = [];
      for (let key in value) {
          arr_item.push(key);
          arr_item.push(value[key]);
          my_arr.push(arr_item);
      }
  }
  
  my_arr.sort((a, b) => a[0] - b[0]);
  
  let res = my_arr[0][1];
  for (let i = 1; i < my_arr.length; i++) {
      res = res + ' ' + my_arr[i][1];
  }

  return res;
}
